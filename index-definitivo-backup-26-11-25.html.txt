<script>
function byId(i){return document.getElementById(i);}
function round(v,d){return Math.round(v*10**d)/10**d;}

const scenarios={
  res:{vMin:1.5,vMax:3.5},
  office:{vMin:2.0,vMax:4.0},
  hospital:{vMin:1.5,vMax:3.0},
  extract:{vMin:5.0,vMax:8.0},
  main:{vMin:6.0,vMax:9.0},
  none:{vMin:null,vMax:null}
};

const stdDiameters=[100,125,150,160,200,250,315,355,400,450,500,560,600];

/* NOTE UNI UNA RIGA */

  const s = byId("scenario").value;
  const note = byId("scenarioNote");

  const notes = {
    res:
      '<span class="info-note-row">Abitazioni â€“ 1.5â€“3.5 m/s â€“ UNI 10339 / UNI EN 15251</span>',
    office:
      '<span class="info-note-row">Uffici â€“ 2â€“4 m/s â€“ UNI 10339 / UNI EN 13779</span>',
    hospital:
      '<span class="info-note-row">Ospedali â€“ 1.5â€“3 m/s â€“ UNI 11425 / UNI EN 13779</span>',
    extract:
      '<span class="info-note-row">Estrazione â€“ 5â€“8 m/s â€“ UNI 10339 / UNI EN 13779</span>',
    main:
      '<span class="info-note-row">Montanti â€“ 6â€“9 m/s â€“ UNI 10339 / UNI EN 13779</span>',
    none:
      '<span class="info-note-row">Nessun limite â€“ verifica disattivata</span>'
  };

  note.innerHTML = notes[s];
}

  updateScenarioNote();

  byId("geom").addEventListener("change",()=>{
    if(byId("geom").value==="rect"){
      byId("rect_fields").style.display="";
      byId("circ_fields").style.display="none";
      byId("dnBlock").style.display="";
    } else {
      byId("rect_fields").style.display="none";
      byId("circ_fields").style.display="";
      byId("dnBlock").style.display="none";
    }
    byId("circOut").style.display="none";
  });

  byId("scenario").addEventListener("change", updateScenarioNote);

  /* ========== AUTO-SELECT su tutti gli input numerici ========== */
  const inputs = document.querySelectorAll('input[type="number"]');
  inputs.forEach(inp=>{

function getTag(v,min,max){
  if(min==null) return {text:"n/d",cls:"tag-na"};
  if(v<min)    return {text:"Bassa",cls:"tag-low"};
  if(v<=max)   return {text:"OK",cls:"tag-ok"};
  return {text:"Alta",cls:"tag-high"};
}

/* =============================================================
   CALCOLO
============================================================= */
function calcola(){
  const Q_raw = parseFloat(byId("flow").value);
  let Q = (!isNaN(Q_raw) && Q_raw>0) ? Q_raw : null;

  const V_raw = parseFloat(byId("velocity").value);
  const V_in = (!isNaN(V_raw) && V_raw>0) ? V_raw : null;

  const geom = byId("geom").value;
  const scen = scenarios[byId("scenario").value];

  if(!Q && !V_in){
    alert("Inserisci almeno Portata Q oppure VelocitÃ  V > 0");
    return;
  }

  let A = null;
  let v = null;

  byId("circOut").style.display="none";
  byId("rectSquareRow").style.display="none";
  byId("suggestCard").style.display="none";

  /* ===================== RETTANGOLARE ===================== */
  if(geom==="rect"){
    const bMm = parseFloat(byId("base").value);
    const hMm = parseFloat(byId("height").value);
    let b = (!isNaN(bMm) && bMm>0) ? bMm/1000 : null;
    let h = (!isNaN(hMm) && hMm>0) ? hMm/1000 : null;

    if(Q){
      const Qs_tmp = Q/3600;

      if(V_in){
        A = Qs_tmp / V_in;
        v = V_in;

        if(b && !h){
          h = A/b;
          byId("height").value = round(h*1000,1);
        } else if(!b && h){
          b = A/h;
          byId("base").value = round(b*1000,1);
        } else if(!b && !h){
          const side = Math.sqrt(A)*1000;
          byId("rectSquareRow").style.display="";
          byId("rectSquareOut").textContent = round(side,1)+" mm";
        } else {
          A = b*h;
          v = Qs_tmp / A;
        }

      } else {
        if(!b || !h){
          alert("Se la velocitÃ  non Ã¨ nota, inserisci base e altezza.");
          return;
        }
        A = b*h;
        v = Qs_tmp / A;
      }

    } else {
      if(!V_in){
        alert("Se la portata non Ã¨ nota, inserisci una velocitÃ  V > 0");
        return;
      }
      if(!b || !h){
        alert("Se la portata non Ã¨ nota, inserisci base e altezza.");
        return;
      }
      A = b*h;
      v = V_in;
    }
  }

  /* ===================== CIRCOLARE ===================== */
  if(geom==="circ"){
    let dMm = parseFloat(byId("diam").value);
    let d   = (!isNaN(dMm) && dMm>0) ? dMm/1000 : null;

    if(Q){
      const Qs_tmp = Q/3600;

      if(V_in){

        if(!d){
          const Aideal = Qs_tmp / V_in;
          const Deq = Math.sqrt(4*Aideal/Math.PI)*1000;

          let best = null, m = 1e9;
          for(const dn of stdDiameters){
            const diff = Math.abs(dn - Deq);
            if(diff < m){ m=diff; best=dn; }
          }

          if(best){
            byId("diam").value = best;
            dMm = best;
            d   = best/1000;
          }
        }

        if(!d){
          A = Qs_tmp / V_in;
          v = V_in;
        } else {
          A = Math.PI*d*d/4;
          v = Qs_tmp / A;
        }

      } else {
        if(!d){
          alert("Se la velocitÃ  non Ã¨ nota, inserisci diametro.");
          return;
        }
        A = Math.PI*d*d/4;
        v = Qs_tmp / A;
      }

    } else {
      if(!V_in){
        alert("Se la portata non Ã¨ nota, inserisci V.");
        return;
      }
      if(!d){
        alert("Se la portata non Ã¨ nota, inserisci diametro.");
        return;
      }
      A = Math.PI*d*d/4;
      v = V_in;
    }

    byId("circOut").style.display="";
    byId("squareSideOut").textContent = round(Math.sqrt(A)*1000,1)+" mm";
  }

  if(!A || !v){
    alert("Errore.");
    return;
  }

  let Qs;
  if(Q && Q>0){
    Qs = Q/3600;
  } else {
    Qs = v*A;
    Q  = Qs*3600;
    byId("flow").value = Math.round(Q);
  }

  /* ===================== RISULTATI GENERALI ===================== */
  byId("areaOut").textContent =
    round(A,4)+" mÂ² ("+round(A*1e6,0)+" mmÂ²)";
  byId("vEffOut").textContent = round(v,2)+" m/s";

  const tag = getTag(v,scen.vMin,scen.vMax);
  byId("vEffTag").textContent = tag.text;
  byId("vEffTag").className = "tag "+tag.cls;

  /* Ã˜ STANDARD rettangolare */
  if(geom==="rect"){
    const Deq = Math.sqrt(4*A/Math.PI)*1000;
    let best = null, m = 1e9;
    for(const dn of stdDiameters){
      if(Math.abs(dn-Deq)<m){m=Math.abs(dn-Deq);best=dn;}
    }

    if(best){
      byId("stdDNOut").textContent = "Ã˜ "+best+" mm";
      byId("stdDNVOut").textContent =
        round(Qs/(Math.PI*(best/1000)**2/4),2)+" m/s";
    } else {
      byId("stdDNOut").textContent  = "â€”";
      byId("stdDNVOut").textContent = "â€”";
    }
  } else {
    byId("stdDNOut").textContent  = "â€”";
    byId("stdDNVOut").textContent = "â€”";
  }

  /* SUGGERIMENTO */
  if(scen.vMax!=null && v>scen.vMax){
    const vT = (scen.vMin+scen.vMax)/2;
    const Anew = Qs/vT;

    const L = Math.sqrt(Anew)*1000;

    let best = null, m = 1e9;
    const DeqNew = Math.sqrt(4*Anew/Math.PI)*1000;
    for(const dn of stdDiameters){
      const diff = Math.abs(dn - DeqNew);
      if(diff < m){
        m = diff;
        best = dn;
      }
    }

    byId("suggText").textContent =
      "VelocitÃ  alta â†’ sezione consigliata: "+
      round(L,1)+" Ã— "+round(L,1)+
      " mm oppure Ã˜ "+best+" mm.";

    byId("suggestCard").style.display="";
  }

  setTimeout(()=>{
    byId("resultsCard").scrollIntoView({behavior:"smooth",block:"start"});
  },120);
}
